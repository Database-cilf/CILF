#DROP ALL TABLES

DROP TABLE MEMBER;
DROP TABLE TAGS;
DROP TABLE RATING;
DROP TABLE PROJECT;
DROP TABLE USER;

#CREATE TABLES

CREATE TABLE USER(  
	id INT NOT NULL AUTO_INCREMENT,
	mongoId CHAR(100) NOT NULL UNIQUE,
	firstName CHAR(100) NOT NULL,
	lastName CHAR(100) NOT NULL,
	created DATE NOT NULL,
	PRIMARY KEY (id)
) ENGINE=InnoDB;

CREATE TABLE PROJECT(
    id INT NOT NULL AUTO_INCREMENT,
    name CHAR(99) NOT NULL,
    description CHAR(255),
    projectUrl CHAR(99) NOT NULL,
    created TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	owner_id CHAR(100) NOT NULL,
    FOREIGN KEY (owner_id) REFERENCES USER(mongoId),
    PRIMARY KEY (id)
) ENGINE=InnoDB;

CREATE TABLE RATING(   
	id INT NOT NULL AUTO_INCREMENT,
	rate INT CHECK( rate BETWEEN 1 AND 10 ),
	proj_id INT NOT NULL,
	user_id CHAR(100) NOT NULL,
	description CHAR(255),
	PRIMARY KEY (id),
	FOREIGN KEY (user_id) REFERENCES USER(mongoId),
	FOREIGN KEY (proj_id) REFERENCES PROJECT(id)
) ENGINE=InnoDB;

CREATE TABLE TAGS(
	id INT NOT NULL AUTO_INCREMENT,
	tag CHAR(99),
	proj_id INT NOT NULL,
	PRIMARY KEY (id),
	FOREIGN KEY (proj_id) REFERENCES PROJECT(id)
) ENGINE=InnoDB;

CREATE TABLE MEMBER(
	id INT NOT NULL AUTO_INCREMENT,
	joined TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	user_id CHAR(100) NOT NULL,
	proj_id INT NOT NULL,
	PRIMARY KEY (id),
	FOREIGN KEY (user_id) REFERENCES USER(mongoId),
	FOREIGN KEY (proj_id) REFERENCES PROJECT(id)
) ENGINE=InnoDB;
